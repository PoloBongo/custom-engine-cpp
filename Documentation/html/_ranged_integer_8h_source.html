94 {
95 static_assert(MIN < MAX);
96 if constexpr (MAX < 0)
97 {
98 // MIN & MAX < 0
99 return static_cast<uint64>(static_cast<int64>(-1) * MIN) - static_cast<uint64>(static_cast<int64>(-1) * MAX);
