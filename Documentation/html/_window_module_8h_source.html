708 void SetWindowMode(const WindowMode& _mode, const bool _bForce = false)
709 {
710 if (_bForce || currentWindowMode != _mode)
711 {
712 currentWindowMode = _mode;
713
714 GLFWmonitor* monitor = glfwGetPrimaryMonitor();
715 if (!monitor)
716 {
717 std::cout << ("Failed to find primary monitor! Can't set window mode\n");
718 return;
719 }
720
721 const GLFWvidmode* video_mode = glfwGetVideoMode(monitor);
722 if (!video_mode)
723 {
724 std::cout << ("Failed to get monitor's video mode! Can't set window mode\n");
725 return;
726 }
727
728 switch (_mode)
729 {
730 case WindowMode::FULLSCREEN:
731 {
732 glfwSetWindowMonitor(window, monitor, 0, 0, video_mode->width, video_mode->height,
733 video_mode->refreshRate);
734 }
735 break;
736 case WindowMode::WINDOWED_FULLSCREEN:
737 {
738 glfwSetWindowMonitor(window, monitor, 0, 0, video_mode->width, video_mode->height,
739 video_mode->refreshRate);
740 lastNonFullscreenWindowMode = WindowMode::WINDOWED_FULLSCREEN;
741 }
742 break;
743 case WindowMode::WINDOWED:
744 {
745 //CHECK(lastWindowedSize.x != 0 && lastWindowedSize.y != 0);
746
747 if (lastWindowedPos.y == 0)
748 // When in windowed mode a y position of 0 means the title bar isn't
749 // visible. This will occur if the app launched in fullscreen since
750 // the last y position to never have been set to a valid value.
751 lastWindowedPos.y = 40;
752
753 glfwSetWindowMonitor(window, nullptr, lastWindowedPos.x, lastWindowedPos.y,
754 lastWindowedSize.x, lastWindowedSize.y, video_mode->refreshRate);
755 lastNonFullscreenWindowMode = WindowMode::WINDOWED;
756 }
757 break;
758 case WindowMode::_NONE:
759 default:
760 {
761 std::cout << ("Unhandled window mode: %u\n", static_cast<uint32_t>(_mode));
762 }
763 break;
764 }
765 }
766 }
